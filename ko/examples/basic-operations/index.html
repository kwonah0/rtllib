    from rtllib import Client

def generate_design_report(verilog_file, output_file):
    """ì„¤ê³„ì˜ JSON ë¦¬í¬íŠ¸ë¥¼ ìƒì„±í•©ë‹ˆë‹¤."""
    with Client() as client:

        client.compile()
        client.elaborate()

        modules = client.get_modules()

        report = {
            "file": verilog_file,
            "modules": []
        }

        for mod in modules:
            module_data = {
                "name": mod['name'],
                "file": mod['file'],
                "statistics": {
                    "ports": len(mod['ports']),
                    "instances": len(mod['instances']),
                    "nets": len(mod['nets'])
                },
                "ports": [
                    {
                        "name": p['name'],
                        "direction": p['direction'],
                        "width": p['width']
                    }
                    for p in mod['ports']
                ]
            }
            report["modules"].append(module_data)

        # ë¦¬í¬íŠ¸ ì‘ì„±
        with open(output_file, "w") as f:
            json.dump(report, f, indent=2)

        print(f"ë¦¬í¬íŠ¸ê°€ {output_file}ì— ì‘ì„±ë¨")
